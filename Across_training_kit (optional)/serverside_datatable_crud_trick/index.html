function settbl(){ /* You may change the link of the server side json URL*/
var link="./serverside_crud_list.php";
SYS_TableServerside(link);
}


function SYS_TableServerside(url){ /* Sets Datatable Appearance*/
SYS_TableStart("#pl_tbl");  
$("#pl_tbl").fadeIn(); 
   $('#pl_tbl').DataTable( {
        "processing": true,
        "serverSide": true,
        "ajax": {
            "url": url+"",
            "dataType": "json",
            "error": function(jqXHR, textStatus, errorThrown){
        $('#pl_tbl tbody').html("No Results Found");
        $('#pl_tbl.dataTables_info').text("Showing 0 to 0 of 0 entries");
           }
        }

    });


}


function SYS_TableStart(tablename){ /* Use this to use multiple datatables at one page*/
$(tablename).hide();  
$(tablename).DataTable().destroy();   
}
